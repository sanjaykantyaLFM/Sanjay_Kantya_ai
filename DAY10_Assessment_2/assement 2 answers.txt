import chromadb
from sentence_transformers import SentenceTransformer

#doc sample
docs = [
    "Python is a popular programming language.",
    "Machine learning helps computers learn from data.",
    "RAG combines retrieval and generation to answer questions."
]

# embedded loading
#model = SentenceTransformer(model)
model = SentenceTransformer("all-MiniLM-L6-v2")

#generate embedding 
embeddings = model.encode(docs).tolist()

#chroma client
client = chromadb.Client()
#chroma collection 
collection = client.create_collection(name="demo_collection")

#Store documents and embeddings
collection.add(
    documents=docs,
    embeddings=embeddings,
    ids=["doc1", "doc2", "doc3"]
)

# Query
query = "what is RAG?"
query_embedding = model.encode(query).tolist()

results = collection.query(
    query_embeddings=[query_embedding],
    n_results=2
)

#print output for two similar documnets
print("Top 2 similar documents:")
for doc in results["documents"][0]:
    print(doc)

ans 12:Minimal langGraph with Two nodes
from typing import TypedDict
from langgraph.graph import StateGraph, START, END

# defining state
class GraphState(TypedDict):
    messages: list

# node 1
def greet(state: GraphState):
    state["messages"].append("Hello!")
    return state

# node 2
def farewell(state: GraphState):
    state["messages"].append("Goodbye!")
    return state

# building here empty graph for the above structure
builder = StateGraph(GraphState)

# adding nodes
builder.add_node("greet", greet)
builder.add_node("farewell", farewell)

# connect node with edges
builder.add_edge(START, "greet")
builder.add_edge("greet", "farewell")
builder.add_edge("farewell", END)

#imp compile the graph now
graph = builder.compile()

# invoke graph
result = graph.invoke({"messages": []})

print(result)

Resources: it means it provide data ( files, database, documents)

from mcp.server.fastmcp import FastMCP

# server initialization
mcp = FastMCP("Word Count Server")

@mcp.tool()
def word_count(text: str) -> int:
    """
    Count the number of words in the given text.
    """
    return len(text.split())

# run server on stdio  bcz stdio is best for local machine
if __name__ == "__main__":
    mcp.run()
