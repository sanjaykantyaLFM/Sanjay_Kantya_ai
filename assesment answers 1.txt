from pydantic import BaseModel, Field

class WeatherReport(BaseModel):
    city: str = Field(min_length=1)
    temperature_c: float = Field(ge=-50, le=60)
    humidity: int = Field(ge=0, le=100) ge for the greateer then and le for the less than
    conditions: list[str]

Valid instance
good = WeatherReport(
    city="Jaipur",
    temperature_c=31.4,
    humidity=42,
    conditions=["sunny", "windy"]
)

 Failing instance
bad = WeatherReport(
    city="",
    temperature_c=3000.0,
    humidity=4999,
    conditions=["snowy"]
)

ans 14 

import { useState } from "react"

function Counter() {
    const [count, setCount] = useState(0)

    return (
        <div>
            <h1>{count}</h1>
            <button onClick={() => setCount(count + 1)}>+1</button>
            <button onClick={() => {
                if(count > 0) {
                    setCount(count - 1)
                }
            }}>-1</button>
            <button onClick={() => setCount(0)}>Reset</button>
        </div>
    )
}

export default Counter

function UserList() {
  const [users, setUsers] = useState([])
  useEffect(() => {
    fetch("https://api.example.com/users")
      .then(r => r.json())
      .then(data => setUsers(data))
  }, []) 
  return (
    <ul>
      {users.map(u => <li key={u.id} > {u.name}</li>  
      )}
    </ul>
  )
}
